স্ট্রিং (String)
Course: Introduction to Programming Language  |  Week: 2  |  Module: 1
স্ট্রিং হলো মূলত ক্যারেক্টারের Array। আমরা জানি একাধিক letter বা ক্যারেক্টার মিলে একটি Word তৈরি হয়, এবং এক বা একাধিক Word মিলে একটি Sentence। প্রোগ্রামিং-এ এই ধরনের Word বা Sentence স্টোর করার জন্য char-এর Array এত বেশি ব্যবহৃত হয় যে এটি একটি আলাদা Data Type হিসেবে পরিচিত হয়ে উঠেছে।
স্ট্রিং ডিক্লেয়ারেশন Declaration
i সংজ্ঞা (Definition)

স্ট্রিং ডিক্লেয়ারেশন মানে হলো প্রোগ্রামকে আগেভাগে জানিয়ে দেওয়া যে, আমি কিছু ক্যারেক্টার (অক্ষর) একসাথে রাখার জন্য মেমোরিতে জায়গা রাখতে চাই। ঠিক যেমন আমরা int বা float ডিক্লেয়ার করি, ঠিক একইভাবে char-এর Array ডিক্লেয়ার করে স্ট্রিং তৈরি করা হয়। স্ট্রিং-এর একদম শেষে সবসময় একটি বিশেষ ক্যারেক্টার '\0' (Null Terminator) থাকে, যা বোঝায় — "এখানেই স্ট্রিং শেষ।"

ii সিনট্যাক্স (Syntax) 
char str[1000];   // 1000 ঘরের একটি স্ট্রিং তৈরি হলো

এখানে 1000 মানে মেমোরিতে ১০০০টি ঘর তৈরি হবে। প্রতিটি ঘরে একটি করে ক্যারেক্টার থাকবে। শেষ ঘরে থাকবে '\0'।

iii বাস্তব জীবনের উদাহরণ
📦
মনে করো তুমি একটি ডিম রাখার ট্রে কিনলে যেখানে ৬টি ডিম রাখা যায়। এখন তুমি "Sifat" নামটি (৫টি অক্ষর) রাখতে চাও। তোমার দরকার ৫টি ঘর অক্ষরের জন্য + ১টি ঘর '\0'-এর জন্য = মোট ৬টি ঘর। তাই char str[6] ডিক্লেয়ার করতে হবে।

iv উদাহরণ কোড
#include <stdio.h>

int main() {
    char str[6];   // 6 ঘরের একটি string ডিক্লেয়ার হলো
    // এখন str ব্যবহারের জন্য প্রস্তুত
    return 0;
}
// কোনো output নেই — শুধু মেমোরি তৈরি হয়েছে
স্ট্রিং ইনিশিয়ালাইজেশন Initialization
i সংজ্ঞা (Definition)

ইনিশিয়ালাইজেশন মানে হলো স্ট্রিং ডিক্লেয়ার করার সময়ই তার ভেতরে মান (ডেটা) রেখে দেওয়া। দুইটি পদ্ধতিতে স্ট্রিং ইনিশিয়ালাইজ করা যায় — (১) প্রতিটি অক্ষর আলাদাভাবে রাখা, অথবা (২) double quotation " " ব্যবহার করে পুরো স্ট্রিং একসাথে রাখা।

ii সিনট্যাক্স (Syntax) 
// পদ্ধতি ১ — প্রতিটি অক্ষর আলাদাভাবে
char str[6] = {'S', 'i', 'f', 'a', 't'};

// পদ্ধতি ২ — double quotation দিয়ে (সহজ উপায়)
char str[6] = "Sifat";

পদ্ধতি ২ বেশি ব্যবহার হয় কারণ এটি সহজ। " "-এর মধ্যে স্ট্রিং লিখলে C নিজেই শেষে '\0' যোগ করে দেয়।

iii বাস্তব জীবনের উদাহরণ
📝
মনে করো তোমার একটি নামের তালিকার খাতা আছে। তুমি দুইভাবে নাম লিখতে পারো — পদ্ধতি ১: প্রতিটি অক্ষর আলাদা ঘরে লেখা: S | i | f | a | t | পদ্ধতি ২: সরাসরি লিখে দেওয়া: "Sifat". দুটোর ফলাফল একই, কিন্তু দ্বিতীয় পদ্ধতিটি অনেক দ্রুত।
iv উদাহরণ কোড
#include <stdio.h>

int main() {
    // পদ্ধতি ১
    char name1[6] = {'S','i','f','a','t'};

    // পদ্ধতি ২
    char name2[6] = "Sifat";

    printf("%s\n", name1);
    printf("%s\n", name2);

    return 0;
}
// Output: Sifat
Sifat
স্ট্রিং প্রিন্টিং Printing
i সংজ্ঞা (Definition)

স্ট্রিং প্রিন্টিং মানে হলো স্ক্রিনে স্ট্রিং-এর ভেতরের অক্ষরগুলো দেখানো। সাধারণ Array-এ লুপ চালিয়ে একটি একটি করে প্রিন্ট করতে হয়, কিন্তু স্ট্রিং-এর একটি বড় সুবিধা হলো — শুধুমাত্র printf() ফাংশনে %s ব্যবহার করলেই পুরো স্ট্রিং একবারে প্রিন্ট হয়ে যায়।

ii সিনট্যাক্স (Syntax) 
printf("%s", str);

// %s  →  স্ট্রিং-এর format specifier
// str →  যে string প্রিন্ট করতে চাই তার নাম
iii বাস্তব জীবনের উদাহরণ
🖨️
মনে করো তুমি একটি ফটোকপি মেশিনে পুরো একটি পাতা দিয়ে দিলে। মেশিন নিজেই পুরো পাতাটি প্রিন্ট করে দেবে — তোমাকে প্রতিটি অক্ষর আলাদা করে বলতে হবে না। printf("%s", str)-ও ঠিক তেমনই কাজ করে — একবার বললেই পুরো স্ট্রিং প্রিন্ট।
iv উদাহরণ কোড
#include <stdio.h>

int main() {
    char str[20] = "Hello, Sifat!";

    printf("%s\n", str);   // পুরো string একবারে প্রিন্ট

    return 0;
}
// Output: Hello, Sifat!
স্ট্রিং ইনপুট Input
i সংজ্ঞা (Definition)

স্ট্রিং ইনপুট মানে হলো ব্যবহারকারীর কাছ থেকে কীবোর্ড দিয়ে একটি শব্দ বা বাক্য নিয়ে প্রোগ্রামে স্টোর করা। scanf() ফাংশন দিয়ে সহজেই স্ট্রিং ইনপুট নেওয়া যায়। তবে মনে রাখতে হবে, scanf("%s") শুধুমাত্র একটি শব্দ নেয় — স্পেস দেখলে থেমে যায়। পুরো বাক্য নিতে fgets() ব্যবহার করতে হয়।

ii সিনট্যাক্স (Syntax) 
// এক শব্দ ইনপুট নিতে
scanf("%s", str);

// পুরো বাক্য ইনপুট নিতে (স্পেস সহ)
fgets(str, sizeof(str), stdin);

scanf-এ int-এর মতো & লাগে না, কারণ স্ট্রিং-এর নামই তার address।

iii বাস্তব জীবনের উদাহরণ
🎤
মনে করো তুমি একটি ফর্ম পূরণ করছো। ফর্মে লেখা আছে "তোমার নাম লেখো।" তুমি নাম লিখলে — সেটাই হলো ইনপুট। প্রোগ্রামে scanf("%s", str) ঠিক এই কাজটি করে — কীবোর্ড থেকে তোমার লেখা নামটি নিয়ে str-এ রেখে দেয়।
iv উদাহরণ কোড
#include <stdio.h>

int main() {
    char name[50];

    printf("তোমার নাম লেখো: ");
    scanf("%s", name);   // ইনপুট নেওয়া হলো

    printf("হ্যালো, %s!\n", name);
    return 0;
}
// Input/Output: তোমার নাম লেখো: Sifat
হ্যালো, Sifat!
বিল্ট-ইন ফাংশন — strlen() String Length
i সংজ্ঞা (Definition)

strlen() ফাংশন একটি স্ট্রিং-এর দৈর্ঘ্য (কতটি ক্যারেক্টার আছে) বের করে দেয়। এটি string.h হেডার ফাইলে থাকে। গুরুত্বপূর্ণ বিষয় হলো — strlen() Null Terminator ('\0') গণনা করে না। তাই "Sifat"-এর length হবে 5, 6 নয়।

ii সিনট্যাক্স (Syntax) 
#include <string.h>   // এই header ফাইল যোগ করতে হবে

strlen(str);   // str-এর দৈর্ঘ্য return করে

// সাধারণত এভাবে ব্যবহার করা হয়:
int len = strlen(str);
iii বাস্তব জীবনের উদাহরণ
📏
মনে করো তুমি একটি সুতার দৈর্ঘ্য মাপতে চাও। তুমি একটি স্কেল নিয়ে মেপে বললে — "এটি ৫ সেমি লম্বা।" strlen() ঠিক এই কাজটি করে — স্ট্রিং-এর মধ্যে কতটি অক্ষর আছে তা গণনা করে জানিয়ে দেয়।

#include <stdio.h>

বিল্ট-ইন ফাংশন — strcpy() String Copy
i সংজ্ঞা (Definition)

strcpy() ফাংশন একটি স্ট্রিং-কে অন্য একটি স্ট্রিং-এ কপি করে। এটিও string.h হেডার ফাইলে থাকে। কপি করার পর destination (যেখানে কপি হবে) স্ট্রিংটি source-এর মতো হয়ে যায়। মনে রাখতে হবে, destination-এর সাইজ অবশ্যই source-এর চেয়ে বড় বা সমান হতে হবে।

ii সিনট্যাক্স (Syntax) 
strcpy(destination, source);

// destination → যেখানে কপি হবে
// source      → যেটা কপি করা হবে
iii বাস্তব জীবনের উদাহরণ
📋
মনে করো তোমার বন্ধুর খাতায় একটি সুন্দর নোট আছে। তুমি সেই নোটটি তোমার খাতায় হুবহু নকল করলে। এখন তোমার খাতায়ও একই নোট আছে। strcpy(str1, str2) ঠিক এই কাজটি করে — str2-এর সব অক্ষর str1-এ কপি করে দেয়।

iv উদাহরণ কোড
#include <stdio.h>
#include <string.h>

int main() {
    char str1[100] = "Sifat";
    char str2[100] = "Pias";

    strcpy(str1, str2);   // str2 → str1-এ কপি হবে

    printf("%s\n", str1);  // str1 এখন "Pias"
    return 0;
}
// Output: Pias
বিল্ট-ইন ফাংশন — strcat() String Concatenate
i সংজ্ঞা (Definition)

strcat() ফাংশন একটি স্ট্রিং-এর শেষে আরেকটি স্ট্রিং জোড়া লাগায়। এটিও string.h-এ থাকে। উদাহরণস্বরূপ, "Sifat" + "Pias" = "SifatPias"। এক্ষেত্রে destination (str1) স্ট্রিং-এর সাইজ এতটাই বড় হতে হবে যাতে দুটো স্ট্রিং একসাথে ধরে।

ii সিনট্যাক্স (Syntax) 
strcat(str1, str2);

// str1-এর শেষে str2 জোড়া লাগবে
// ফলাফল str1-এ জমা হবে
iii বাস্তব জীবনের উদাহরণ

iv উদাহরণ কোড
#include <stdio.h>
#include <string.h>

int main() {
    char str1[100] = "Sifat";   // বড় সাইজ দিতে হবে!
    char str2[100] = "Pias";

    strcat(str1, str2);   // str1-এর শেষে str2 যুক্ত হবে

    printf("%s\n", str1);
    return 0;
}
// Output: SifatPias
⚠️ সাবধান: str1-এর সাইজ অবশ্যই str1 ও str2-এর মোট অক্ষর সংখ্যার চেয়ে বড় হতে হবে। নইলে মেমোরি সমস্যা হতে পারে!
